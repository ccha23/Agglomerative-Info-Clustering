6  /*
7  Find the point in the base polytope of the normalized version of a submodular function
8  that minimizes the weighted sum of coordinates.
9  @param f The submodular function
10  @param w The weight vector with length equal to that of the ground set of f.
11  @return The vector x in B(f-f(emptySet)) such w'x is minimized.
12  */
13  Eigen::VectorXd edmonds_greedy(const SF &f, const Eigen::VectorXd &w) {
14  size_t n = f.size();
15  if (n != w.size())
16  throw std::runtime_error("w must have the same size as the ground set of f.");
17  std::vector<size_t> idx(n);
18  Eigen::VectorXd x(n);
19  std::iota(idx.begin(), idx.end(), 0);
20  sort(idx.begin(), idx.end(), [&](size_t i1, size_t i2) {return w[i1] < w[i2]; });
21  std::vector<size_t> B;
22  double F = f(B);
23  for (auto i : idx) {
24  x[i] = -F;
25  B.push_back(i);
26  F = f(B);
27  x[i] += F;
28  }
29  return x;
30  }

32  /*
33  Compute the minimum norm base in the base polytope of the normalized version of a submodular function
34  with specified tolerance and precision. N.b., set function tolerance to 1E-6 or above due to the numerical issue with Eigen.
35  @param f The submodular function
36  @param fn_tol Function Tolerance
37  @param eps Precision
38  @return The minimum norm point of the base B(f-f(emptySet)).
39  */
40  Eigen::VectorXd min_norm_base(const SF &f, double fn_tol, double eps) {
41  size_t n = f.size();
42  Eigen::VectorXd x;
43  if (n < 1)
44  return x;
45  x = Eigen::VectorXd::Ones(n);
46  if (n == 1) {
47  std::vector<size_t> V(n);
48  std::iota(V.begin(), V.end(), 0);
49  return x*f(V);
50  }
51  // Step 0 (Initialize a trivial corral Q and x=Nr Q)
52  x = edmonds_greedy(f, x);
53  //std::cout << f(std::vector<size_t> {0, 1}) - f(std::vector<size_t> {0}) << std::endl;
54  //std::cout << x.transpose() << std::endl;
55  Eigen::MatrixXd Q(n, n + 1);
56  Q.col(0) = x;
57  size_t numCorral = 1; // Corral: First numCorral columns of Q
58 
59  Eigen::VectorXd w = Eigen::VectorXd::Zero(n + 1);
60  w(0) = 1; // maintain x == Q * w
61 
62  // initialization for adaptive computation of the projection to affine hull of the corral
63  Eigen::VectorXd e = Eigen::VectorXd::Ones(n + 1);
64  Eigen::MatrixXd L = Eigen::MatrixXd::Zero(n + 1, n + 1);
65  L(0, 0) = std::sqrt(1 + x.dot(x)); // maintain L*L.transpose() == e*e.transpose() + Q.transpose() * Q
66  // for the first numCorral rows and columns.
67  double old_first_order_opt,first_order_opt = INFINITY;
68  size_t stuck_count;
69  while (1) {
70  // Step 1 (Major cycle: move towards origin along the direction of x)
71  Eigen::VectorXd q = edmonds_greedy(f, x);
72  old_first_order_opt = first_order_opt;
73  first_order_opt = std::abs(x.dot(q) - x.dot(x));
74  //log<LOG_INFO>(L"[min_norm_base] First order optimality : %1% ") % first_order_opt;
75  // stopping criteria
76  if (first_order_opt < fn_tol) {
77  return x; // x separates the base polytope from the origin and is therefore the minimum norm point.
78  }
79  if (old_first_order_opt <= first_order_opt) { // stuck
80  stuck_count++;
81 // std::cout << "stuck : " << stuck_count << " " << first_order_opt << std::endl;
82  if (stuck_count > 100) { // avoid getting stuck for too long
83  std::cerr << "Unstuck at gap to optimality: " << first_order_opt << std::endl;
84  return x;
85  }
86  }
87  else {
88  stuck_count = 0;
89  }
90  {
91  // Step 2 (Major cycle: project to affine hull)
92  auto L_ = L.topLeftCorner(numCorral, numCorral).triangularView<Eigen::Lower>();
93  auto Q_ = Q.topLeftCorner(n, numCorral);
94  auto e_ = e.topRows(numCorral);
95  Eigen::VectorXd r = L_.solve(e_ + Q_.transpose() * q);
96  L.block(numCorral, 0, 1, numCorral) = r.transpose();
97  L(numCorral, numCorral) = 1 + q.dot(q) - r.dot(r);
98  //std::cout << "Optimality test:" << std::endl;
99  //std::cout << "x : " << x.transpose() << std::endl;
100  //std::cout << "q : " << q.transpose() << std::endl;
101  //std::cout << "Q : " << std::endl << Q_ << std::endl;
102  if (1 + q.dot(q) - r.dot(r) < 0) {
103  //std::cout << "Update L to include q:" << std::endl;
104  //std::cout << "L: " << std::endl << (Eigen::MatrixXd) L_ << std::endl;
105  //std::cout << "r: " << r.transpose() << std::endl;
106  throw std::runtime_error("Try to set a higher functional tolerance.");
107  }
108  L(numCorral, numCorral) = sqrt(1 + q.dot(q) - r.dot(r));
109  Q.col(numCorral++) = q;
110  }
111  int minor_cycle_count = 0; // count number of iterations of minor cycles in iteration of the major cycle
112  while (1) {
113  minor_cycle_count = minor_cycle_count + 1;
114  auto L_ = L.topLeftCorner(numCorral, numCorral).triangularView<Eigen::Lower>();
115  auto Q_ = Q.topLeftCorner(n, numCorral);
116  auto e_ = e.topRows(numCorral);
117  Eigen::VectorXd v = L_.transpose().solve(L_.solve(e_));
118  v = (e_ * e_.transpose() + Q_.transpose() * Q_).llt().solve(e_); // cc
119  v = v / e_.dot(v);
120  //std::cout << "Affine projection: origin onto aff(Q) " << std::endl;
121  //std::cout << "L : " << std::endl << (Eigen::MatrixXd)L_ << std::endl;
122  //std::cout << "Q : " << std::endl << Q_ << std::endl;
123  //std::cout << "v : " << v.transpose() << std::endl;
124  bool minor_cycle = false;
125  {
126  size_t i = 0;
127  while (!minor_cycle && i < numCorral)
128  minor_cycle = (v(i++) <= eps);
129  }
130  auto w_ = w.topRows(numCorral);
131  if (!minor_cycle) {
132  // continue with Main Cycle
133  w_ = v;
134  x = Q_*w_;
135  //log<LOG_INFO>(L"[min_norm_base] # minor cycles : %1% ") % minor_cycle_count;
136  break;
137  }
138  // Step 3: Minor cycle
139  double theta = 1;
140  for (size_t i = 0; i < numCorral; i++) {
141  if (w_(i) - v(i)>= eps) {
142  theta = std::min(theta, w_(i) / (w_(i) - v(i)));
143  }
144  }
145  w_ = (1 - theta) * w_ + theta *v;
146  size_t i_ = 0;
147  std::vector<bool> toDelete(numCorral);
148  for (size_t i = 0; i < numCorral; i++) {
149  if ((toDelete[i] = (w_(i) <= eps))) {
150  for (size_t j = i; j < numCorral - 1; j++) {
151  // Maintain L triangular after deletion
152  double a = L(j + 1, j), b = L(j + 1, j + 1);
153  double c = sqrt(a*a + b*b);
154  Eigen::Matrix2d T;
155  T << a / c, -b / c, b / c, a / c;
156  L.block(j + 1, j, numCorral - j, 2) *= T;
157  }
158  }
159  }
160  // perform deletion
161  for (size_t i = 0; i < numCorral; i++) {
162  if (!toDelete[i]) {
163  if (i_<i) {
164  w_(i_) = w_(i);
165  Q.col(i_) = Q.col(i);
166  L.row(i_) = L.row(i);
167  } // else, i.e., i_=i : no need to move
168  i_++;
169  } // else: delete ith corral
170  }
171  numCorral = i_;
172  x = Q.topLeftCorner(n, numCorral)*w.topRows(numCorral);
173  }
174  }
175  return x;
176  }
