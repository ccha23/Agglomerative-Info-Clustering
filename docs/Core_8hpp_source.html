183  bool merge(size_t i, size_t j, double gamma) {
184  size_t iroot = find(i);
185  size_t jroot = find(j);
186  if (iroot == jroot || // i and j already merged
187  not (this->gamma.empty() ||
188  this->gamma.back() >= gamma)) // gamma is too large
189  return false;
190  if (this->gamma.empty() || this->gamma.back() > gamma)
191  this->gamma.push_back(gamma); // update the list of distinct critical values
192  if (rank[iroot] <= rank[jroot]) {

194  parent[iroot] = jroot;
195  children[jroot].push_back(iroot);
